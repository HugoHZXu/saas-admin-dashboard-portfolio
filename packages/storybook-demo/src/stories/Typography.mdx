import { Meta } from '@storybook/addon-docs/blocks';
import * as Typography from 'hugo-ui/styles/theme';
import { StorybookSetStyle, StorybookSetRowStyle, StorybookFontStyle } from './assets/styles';

export const sampleText = 'The quick brown fox jumps over the lazy dog';

export const typesetSizeStyle = {
  marginRight: 30,
  ...StorybookFontStyle,
};

export const typesetTextStyle = {
  overflow: 'hidden',
  whiteSpace: 'nowrap',
  textOverflow: 'ellipsis',
};

export const Typeset = (styles) => {
  return (
    <div style={{ ...StorybookSetStyle }}>
      {styles.map((style) => (
        <div style={{ ...StorybookSetRowStyle }}>
          <span style={{ ...typesetSizeStyle }}>{style.fontSize}</span>
          <span style={{ ...typesetTextStyle, ...style }}>{sampleText}</span>
        </div>
      ))}
    </div>
  );
};

# Typography
