import { Meta, Markdown } from '@storybook/addon-docs/blocks';
import { StorybookSetStyle, StorybookSetRowStyle, StorybookFontStyle } from './assets/styles';
import Card from '@mui/material/Card';
import {
  NO_SHADOW,
  INNER_SHADOW,
  LIGHT_SHADOW,
  MEDIUM_SHADOW,
  HEAVY_SHADOW,
} from 'hugo-ui/styles/theme';

export const elevationSetStyle = {
  ...StorybookSetStyle,
  background: '#F2F2F2',
  padding: '30px 20px',
};

export const elevationRowStyle = {
  display: 'flex',
  alignItems: 'flex-start',
};

export const elevationInfoStyle = {
  margin: '0 0 0 30px',
  listStyle: 'none',
};

export const listItem = (content) => {
  return <li style={{ ...StorybookFontStyle }}> {content} </li>;
};

export const ElevationSet = (style) => {
  const boxShadow = style.boxShadow.split(' ')
  if (boxShadow[0] === 'inset') boxShadow.shift()
  const rgba = style.boxShadow.substring(style.boxShadow.indexOf('rgb'))
  const alpha = rgba.substring(rgba.lastIndexOf(','), rgba.length - 1)
  const color = rgba.replace(alpha, '')

return (

<div style={{ ...elevationSetStyle }}>
  <div style={{ ...elevationRowStyle }}>
    <Card style={{ width: 200, height: 200, ...style }} />
    {boxShadow[0] !== 'none' && (
      <ul style={{ ...elevationInfoStyle }}>
        {listItem(`X: ${boxShadow[0] || ''} `)}
        {listItem(`Y: ${boxShadow[1] || ''} `)}
        {listItem(`Blur: ${boxShadow[2] || ''} `)}
        {listItem('Spread: 0')}
        {listItem(`Color: ${color || ''} `)}
        {listItem(`Opacity: ${alpha.substring(2) || ''} `)}
      </ul>
    )}
  </div>
</div>
) }

# Elevation

> These stylings use box-shadows to make an elements appear elevated in front of another

<br />

## No Shadow

**Style:** `NO_SHADOW`

**Typical uses:** Inline controls (e.g., text links), low-emphasis actions, flat surfaces without elevation

<div>{ElevationSet(NO_SHADOW)}</div>

<br />

## Inner Shadow

**Style:** `INNER_SHADOW`

**Typical uses:** Pressed / inset states (e.g., segmented controls, toggles, active chips) to indicate a recessed surface

<div>{ElevationSet(INNER_SHADOW)}</div>

<br />

## Light Shadow

**Style:** `LIGHT_SHADOW`

**Typical uses:** Small surface elements (e.g., compact cards, tooltips-like mini surfaces) where subtle separation is needed

<div>{ElevationSet(LIGHT_SHADOW)}</div>

<br />

## Medium Shadow

**Style:** `MEDIUM_SHADOW`

**Typical uses:** Standard surfaces (e.g., cards, panels) and overlay containers (e.g., dialogs) in the primary content layer

<div>{ElevationSet(MEDIUM_SHADOW)}</div>

<br />

## Heavy Shadow

**Style:** `HEAVY_SHADOW`

**Typical uses:** High-priority overlays (e.g., menus, popovers, dropdowns) and topmost layers that must visually float above content

<div>{ElevationSet(HEAVY_SHADOW)}</div>
